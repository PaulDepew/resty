### Setup

Clone the Git repository
Install Dependencies
   - "eslint-plugin-react": "^7.20.3",
   - "node-sass": "^4.14.1",
   - "react": "^16.13.1",
   - "react-dom": "^16.13.1",
   - "react-scripts": "^3.4.1"

Assign Scripts:
  - "start": "react-scripts start",
  - "build": "react-scripts build",
  - "test": "react-scripts test --env=jsdom",
  - "eject": "react-scripts eject"

Run npm start

Enter a Url

Click on an Action
